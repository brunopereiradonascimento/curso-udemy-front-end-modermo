//Instalação - Globalmente
npm install gulp-cli -g ou npm install -g gulp

//Instalação local
npm install gulp --save-dev ou npm i gulp -D

//Criação do arquivo de configuração
touch gulpfile.js

//Desintalando. Ex.: node-sass
npm uninstall node-sass --save-dev ou
npm un node-sass --save-dev

//Instalando o gulp-sass
npm i gulp-sass -D

//Rodando o gulp - sass
npm run gulp sass

//Browser Sync - instalação
npm i browser-sync -D

//Browser Sync - executar
npm run gulp servidor

//Gulp file include
npm i gulp-file-include -D

//Gulp clean
npm i gulp-clean -D

//Gulp Auprefixer - adciona "-webkits-" no sass/css
npm i gulp-autoprefixer -D

//Gulp Uncss - limpa css não utilizado
npm i gulp-uncss -D

//Gulp minifica PNG, JPEG, GIF and SVG
npm i -D gulp-imagemin

//Gulp minifica css
npm i -D gulp-cssnano

//Gulp concatena arquivos
npm i -D gulp-concat

//Gulp minifica javascript
npm i -D gulp-uglify

//Gulp renomeia arquivos
npm i -D gulp-rename


/*********/
Ao criar uma task('default') com o nome default, é possível roda-la somente com o comando npm run gulp
